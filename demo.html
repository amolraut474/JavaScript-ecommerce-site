`; productItem.addEventListener('click', () => showSingleProduct(id)); return productItem; }; // Function to show single product page const showSingleProduct = productId => { const product = products.find(p => p.id === productId); productName.textContent = product.name; productDescription.textContent = product.description; singleProduct.style.display = 'block'; }; // Function to hide single product page const hideSingleProduct = () => { singleProduct.style.display = 'none'; }; // Populate the product list products.forEach(product => { const productItem = createProductItem(product.id, product.name, product.description); productList.appendChild(productItem); });
