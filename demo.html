#quickView, #singleProduct { position: fixed; top: 50%; left: 50%; transform: translate(-50%, -50%); background-color: white; padding: 20px; box-shadow: 0 0 10px rgba(0, 0, 0, 0.1); display: none; } .productItem { margin: 10px; padding: 10px; border: 1px solid #ddd; cursor: pointer; } .productItem:hover { background-color: #f0f0f0; } const productList = document.getElementById('productList'); const quickView = document.getElementById('quickView'); const singleProduct = document.getElementById('singleProduct'); const productName = document.getElementById('productName'); const productDescription = document.getElementById('productDescription'); // Sample product data const products = [ { id: 1, name: 'Product 1', description: 'Description for Product 1' }, { id: 2, name: 'Product 2', description: 'Description for Product 2' }, // Add more products as needed ]; // Function to dynamically create product items const createProductItem = (id, name) => { const productItem = document.createElement('div'); productItem.classList.add('productItem'); productItem.dataset.productId = id; productItem.textContent = name; productItem.addEventListener('click', () => showQuickView(id)); return productItem; }; // Function to show quick view on click const showQuickView = productId => { const product = products.find(p => p.id === productId); productName.textContent = product.name; productDescription.textContent = product.description; quickView.style.display = 'block'; }; // Function to show single product page const showSingleProduct = () => { singleProduct.style.display = 'block'; quickView.style.display = 'none'; }; // Function to hide single product page const hideSingleProduct = () => { singleProduct.style.display = 'none'; }; // Populate the product list products.forEach(product => { const productItem = createProductItem(product.id, product.name); productList.appendChild(productItem); });
